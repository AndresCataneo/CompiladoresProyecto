// Programa de ejemplo que cumple las regex dadas

{ 
  // variables (Ids según: [a-z]+[A-Z])
  Bsum := 0;
  Cn := 10;
  Ai := 0;
  Dflag := true;

  // ciclo for: acumulador de 0 a nC
  for (Ai := 0; Ai <= Cn; Ai := Ai + 1) {
    Bsum := Bsum + Ai;
  }

  // uso de if/then/else con operadores booleanos y comparaciones
  if (Bsum == 55 & Dflag == true) then {
    // comentario: mostramos resultado si la condición es verdadera
    Eresultado := Bsum;
  } else {
    Eresultado := -1;
  }

  // while con operador lógico NOT (!)
  Fcontador := 5;
  while (Fcontador <= 0) {
    Fcontador := Fcontador - 1;
    Dflag := !Dflag;
  }

  // operación aritmética mixta
  Gvalor := (Eresultado * 2) / (Cn + 1);

  // fin del bloque
}
